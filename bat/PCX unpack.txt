@echo off
setlocal enabledelayedexpansion

TITLE SkullMod - PCX unpack
echo PCX unpack
echo.


set argCount=0
set fileParams=-files
REM A 'call' is required for the parameter to be evaluated correctly
FOR %%p IN (%*) DO (
    set /A argCount+=1
    call :concat %%p
)

echo Number of files to process: %argCount%
echo.

if %argCount% == 0 (
    echo No files given, drag and drop the .pcx files on this file
    echo.
) else (
    REM change directory to local directory of bat file to start SkullMod
    pushd "%~dp0"
    SkullMod.exe -do unpack -pcx %fileParams%
    echo.
)

pause
goto :eof

:concat
set fileParams=%fileParams% %1

:eof
